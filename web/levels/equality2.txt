Proposition : Type;
Equals : Type -> Type -> Proposition;

reflexive : (x : Type) -> Equals x x;
symmetric : (x : Type) -> (y : Type) -> Equals x y -> Equals y x;
transitive : (x : Type) -> (y : Type) -> (z : Type) -> Equals x y -> Equals y z -> Equals x z;
